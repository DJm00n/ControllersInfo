typedef struct
{

  uint16_t GD_GamePadX;                              // Usage 0x00010030: X, Value = 0 to -1, Physical = Value x --1
                                                     // Collection: CA:GamePad
  uint8_t  GD_GamePadZ;                              // Usage 0x00010032: Z, Value = 0 to 255, Physical = Value x -1 / 255
